    class Config:
        extra = "forbid"

[docs]    @validator('url')
    def validate_url(cls, value):
        """ESS.url validator"""
        if '.' not in value:
            raise ValueError('url invalid (expected a ".")')
        if ' ' in value:
            raise ValueError('url invalid (no spaces allowed)')
        return value
